Maintenance Scheduling

Algorithm: Priority Queue Scheduling ‚Äî O(n log n)

#include <iostream>
#include <queue>
using namespace std;

int main() {
    priority_queue<int> q;

    q.push(3);
    q.push(1);
    q.push(2);

    while(!q.empty()) {
        cout << "Task priority " << q.top() << endl;
        q.pop();
    }

    cout << "Maintenance done" << endl;
}
