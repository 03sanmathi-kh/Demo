Intelligent Traffic Density Management System

Problem

The city faces uneven traffic distribution because vehicles crowd main roads while other routes stay empty. Manual traffic control fails during peak hours, leading to congestion, delays, and inefficient EV flow.

Solution

A real-time automated traffic density system that uses sensors, cameras, and algorithms to monitor road load, rank congestion levels, and automatically adjust signal timings.

Data Structures Used

  • Array â†’ Store live sensor data (vehicle count, speed).
  • Max Heap â†’ Identify most congested road instantly.
  • Queue â†’ Manage vehicles approaching each signal.
  • Graph â†’ Represent entire road network for cross-road density checks.

Algorithms Used

  • Merge Sort â†’ Sort road densities for ranking.
  • Heap Operations â†’ Prioritize which road gets green signal first.
  • BFS â†’ Spread density updates across nearby intersections.

Step-by-Step Process

1. Sensors send vehicle counts â†’ stored in array.

2. Merge Sort arranges roads from most to least crowded.

3. Max Heap picks the top 3 highest-density roads.

4. BFS sends density updates to all connected intersections (graph).

5. Heap-based prioritization assigns green time to the busiest lanes.

6. Signal controller updates timings in real time.

7. Dashboard shows congestion map to users.

 {
 int p = mid - l + 1;   
    int q = r - mid;         
 int B[p], C[q];          
 // Copy A[l..mid] â†’ B[]
    for(int i = 0; i < p; i++)
        B[i] = A[l + i];
 // Copy A[mid+1..r] â†’ C[]
    for(int j = 0; j < q; j++)
        C[j] = A[mid + 1 + j];
 int i = 0, j = 0, k = l;
 // Merge (DESCENDING ORDER)
    while(i < p && j < q)
             {
        if(B[i] >= C[j])
     {
       A[k] = B[i];
            i++;
        } 
     else
     {
     A[k] = C[j];
            j++;
     }
        k++;
  } // Remaining elements
    while(i < p)
        A[k++] = B[i++];
    
    while(j < q)
        A[k++] = C[j++];
}
void mergeSort(int A[], int l, int r) {
    if(l < r) {
        int mid = (l + r) / 2;
        mergeSort(A, l, mid);
        mergeSort(A, mid + 1, r);
        mergeArray(A, l, mid, r);
    }
}

void heapify(int h[], int n, int i)
     {
    int k = i, v = h[i];
    bool heap = false;
    while(!heap && (2*k + 1) < n)
  {
     int j = 2*k + 1;
        if(j + 1 < n && h[j] < h[j+1])
            j++;
        if(v >= h[j])
            heap = true;
        else {
            h[k] = h[j];
            k = j;
        }
    }
    h[k] = v;
}

Problem

using namespace std;

void heapify(vector<int> &H, int n, int i){
    int smallest=i, l=2*i+1, r=2*i+2;
    if(l<n && H[l]<H[smallest]) smallest=l;
    if(r<n && H[r]<H[smallest]) smallest=r;
    if(smallest!=i){
        swap(H[i], H[smallest]);
        heapify(H,n,smallest);
    }
}

void buildHeap(vector<int> &H){
    int n=H.size();
    for(int i=n/2-1;i>=0;i--) heapify(H,n,i);
}

int main(){
    vector<int> H={30,10,20,40,50};
    buildHeap(H);
    for(int x:H) cout << x << " ";
}

#include <iostream>
#include <vector>
#include <queue>
using namespace std;

void BFS(int start, vector<vector<int>> &graph, vector<bool> &visited){
    queue<int> q;
    q.push(start);
    visited[start]=true;
    while(!q.empty()){
        int u=q.front(); q.pop();
        cout << "Node " << u << " needs repair" << endl;
        for(int v:graph[u]){
            if(!visited[v]){
                visited[v]=true;
                q.push(v);
            }
        }
    }
}

int main(){
    vector<vector<int>> g={{1,2},{0,3},{0},{1}};
    vector<bool> vis(4,false);
    BFS(0,g,vis);
}

#include <iostream>
using namespace std;

void quicksort(int arr[], int l, int r){
    if(l>=r) return;
    int pivot=arr[r], i=l-1;
    for(int j=l;j<r;j++){
        if(arr[j]<pivot){
            i++; swap(arr[i],arr[j]);
        }
    }
    swap(arr[i+1],arr[r]);
    quicksort(arr,l,i);
    quicksort(arr,i+2,r);
}

int main(){
    int arr[]={50,10,40,20,30};
    quicksort(arr,0,4);
    for(int x:arr) cout << x << " ";
}

#include <iostream>
#include <vector>
using namespace std;

void DFS(int u, vector<vector<int>> &graph, vector<bool> &visited){
    visited[u]=true;
    cout << "Node " << u << " needs repair" << endl;
    for(int v:graph[u]){
        if(!visited[v]) DFS(v,graph,visited);
    }
}

int main(){
    vector<vector<int>> g={{1,2},{0,3},{0},{1}};
    vector<bool> vis(4,false);
    DFS(0,g,vis);
}
